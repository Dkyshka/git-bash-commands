# Git Bash Commands
```
clear                  			# очистить консоль

Навигация
pwd                    			    # показать текущий каталог
ls                     			    # показывает файлы в текущий папке
cd ..                  			    # вернутся назад
cd ../..               			    # выйти на 2 уровня вверх


Создание каталогов
mkdir app1             			    # создать папку
mkdir app1 app2        			    # создать несколько папок
mkdir app1/{css,js}    			    # создать несколько вложенных папок
cd !$                  			    # перейти в только что созданную папку

Создание файлов
touch index.html                # создать файл
touch app/{main.css,main.js}    # создать сразу несколько файлов

Чтение файла
cat                             # чтение файла

Удаление файлов
rm index.html                   # удаление файла
rm -r app                       # удаление папки с файлами внутри
rm *.txt                        # удаление файлов с расширением .txt

Открытие файла/каталога
start index.html                # открыть файл в приложении по умолчанию или каталог

Перемещение файлов
mv text.txt app                 # переместить один файл в папку app
mv app/text.txt ../             # переместить на уровень выше находясь на этом же уровне
mv app/text.txt ./              # переместить на уровень выше находясь на выше уровне
mv app/*.* app2                 # переместить все файлы из папки app в app2
mv *.txt *.tmp *.old /tmp       # переместить определенные форматы файлов

```
